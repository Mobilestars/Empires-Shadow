add_namespace = ES_war_support
country_event = {
	id = ES_war_support.1
	hidden = yes
	is_triggered_only = yes
	immediate = {
		if = {
			limit = { has_idea = ES_ws_national_fervor }
			remove_ideas = ES_ws_national_fervor
		}
		if = {
			limit = { has_idea = ES_ws_wavering_resolve }
			remove_ideas = ES_ws_wavering_resolve
		}
		if = {
			limit = { has_idea = ES_ws_crumbling_morale }
			remove_ideas = ES_ws_crumbling_morale
		}
		if = {
			limit = { has_idea = ES_ws_brink_of_collapse }
			remove_ideas = ES_ws_brink_of_collapse
		}
		if = {
			limit = { has_war_support = 1.0 }
			add_ideas = ES_ws_national_fervor
		}
		else_if = {
			limit = {
				NOT = { has_war_support = 0.50 }
				has_war_support = 0.30
			}
			add_ideas = ES_ws_wavering_resolve
		}
		else_if = {
			limit = {
				NOT = { has_war_support = 0.30 }
				has_war_support = 0.10
			}
			add_ideas = ES_ws_crumbling_morale
		}
		else_if = {
			limit = { NOT = { has_war_support = 0.10 } }
			add_ideas = ES_ws_brink_of_collapse
		}
		if = {
			limit = {
				has_idea = ES_echoes_of_the_trenches
				has_war_support = 1.0
			}
			remove_ideas = ES_echoes_of_the_trenches
		}
	}
	option = {
		name = ES_war_support.1.a
	}
}
